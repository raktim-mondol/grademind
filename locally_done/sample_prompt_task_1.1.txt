    Parameters:
    - X_train (DataFrame): Training features.
    - X_test (DataFrame): Test features.

    Returns:
    - X_train_filled (DataFrame): Training features with no missing values.
    - X_test_filled (DataFrame): Test features with no missing values.
    """
        # copy
    Xtr, Xte = X_train.copy(), X_test.copy()


    for df in (Xtr, Xte):
        df.replace(["?", " ?", "", "NA", "NaN", "nan"], np.nan, inplace=True)

    # devide
    num_cols = Xtr.select_dtypes(include=[np.number]).columns.tolist()
    cat_cols = Xtr.columns.difference(num_cols).tolist()

    # Numerical column: Mean interpolation
    if num_cols:
        imp_num = SimpleImputer(strategy="mean")
        Xtr[num_cols] = imp_num.fit_transform(Xtr[num_cols])
        Xte[num_cols] = imp_num.transform(Xte[num_cols])

    #Category column: Mode interpolation
    if cat_cols:
        imp_cat = SimpleImputer(strategy="most_frequent")
        Xtr[cat_cols] = imp_cat.fit_transform(Xtr[cat_cols])
        Xte[cat_cols] = imp_cat.transform(Xte[cat_cols])

    return Xtr, Xte
